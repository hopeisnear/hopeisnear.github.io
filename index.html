No parameters and empty option

Selected value in model: {{country || 'undefined'}}
Markup
<select data-jq-select2="" data-ng-model="country" data-ng-options="country.code as country.name for country in countries" placeholder="Select country"> <option></option> </select>
Data in format:
$scope.countries = [ {name: 'Afghanistan', code: 'AF'}, {name: 'Albania', code: 'AL'}, ... {name: 'Zimbabwe', code: 'ZW'} ];

Preselected value and no empty option available

Selected value in model: {{preselectedCountry || 'undefined'}}
Markup
<select class="jq-select2" data-jq-select2="" data-ng-model="preselectedCountry" data-ng-options="country.name for country in countries" placeholder="Select country"> </select>
Data in format:
$scope.preselectedCountry = $scope.countries[5];

Dynamic loaded data from json with custom select 2 options

Use load states button to load data from json file.
States loaded
Selected value in model: {{state || 'undefined'}}
Markup
<select class="jq-select2" data-jq-select2="statesSelect2Options" data-ng-model="state" data-ng-options="key as value for (key, value) in states" placeholder="Select state"> <option></option> </select>
Data in format:
$scope.states = { "AL": "Alabama", "AK": "Alaska", "AS": "American Samoa", "AZ": "Arizona", ... "WY": "Wyoming" } $scope.statesSelect2Options = { minimumResultsForSearch: -1 };
